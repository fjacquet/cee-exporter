Target Close() (CRITICAL ordering — see Pitfalls):
```go
func (w *Writer) Close() error {
    close(w.done)   // 1. signal goroutine
    w.wg.Wait()     // 2. wait — WITHOUT holding any lock
    w.mu.Lock()     // 3. safe to acquire now
    defer w.mu.Unlock()
    if len(w.records) == 0 {
        return nil
    }
    return w.flushToFile() // 4. final flush
}
```

Note: close(w.done) panics if called twice. Use sync.Once in the Writer struct if multiple Close() calls are possible in tests. The backgroundLoop goroutine is started only when cfg.FlushIntervalSec > 0.

    ```go
    // backgroundLoop runs as a goroutine when FlushIntervalSec > 0.
    // It calls flushToFile() under w.mu on every ticker interval.
    // It exits when w.done is closed.
    func (w *Writer) backgroundLoop() {
        defer w.wg.Done()
        ticker := time.NewTicker(time.Duration(w.cfg.FlushIntervalSec) * time.Second)
        defer ticker.Stop()
        for {
            select {
            case <-ticker.C:
                w.mu.Lock()
                if len(w.records) > 0 {
                    _ = w.flushToFile()
                }
                w.mu.Unlock()
            case <-w.done:
                return
            }
        }
    }
    ```

    ```go
    func (w *Writer) Close() error {
    ```
