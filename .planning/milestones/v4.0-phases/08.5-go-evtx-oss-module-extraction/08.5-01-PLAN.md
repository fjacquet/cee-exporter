    ```go
    package evtx

    import (
        "fmt"
        "log/slog"
        "os"
        "path/filepath"
        "sync"
    )

    type Writer struct {
        mu       sync.Mutex
        path     string
        records  []byte
        recordID uint64
        firstID  uint64
    }

    func New(path string) (*Writer, error) {
        // validate non-empty path, MkdirAll parent, return Writer{path, recordID:1, firstID:1}
    }

    // WriteRaw appends a pre-encoded BinXML payload (NOT yet wrapped with record header).
    // go-evtx wraps it via wrapEventRecord with w.recordID and time.Now() as timestamp.
    // Use one or the other (WriteRaw/WriteRecord), not both in the same session.
    func (w *Writer) WriteRaw(chunk []byte) error {
        w.mu.Lock()
        defer w.mu.Unlock()
        ts := toFILETIME(time.Now())
        rec := wrapEventRecord(w.recordID, ts, chunk)
        w.records = append(w.records, rec...)
        w.recordID++
        return nil
    }

    func (w *Writer) WriteRecord(eventID int, fields map[string]string) error {
        w.mu.Lock()
        defer w.mu.Unlock()
        binXMLChunkOffset := evtxRecordsStart + uint32(len(w.records)) + evtxRecordHeaderSize
        payload := buildBinXML(eventID, fields, binXMLChunkOffset)
        ts := toFILETIME(parseTimeCreated(fields))
        rec := wrapEventRecord(w.recordID, ts, payload)
        w.records = append(w.records, rec...)
        w.recordID++
        if len(w.records) > chunkFlushThreshold {
            slog.Warn("go_evtx_chunk_boundary_reached", "path", w.path, "buffered_bytes", len(w.records))
        }
        return nil
    }

    func (w *Writer) Close() error {
        w.mu.Lock()
        defer w.mu.Unlock()
        if len(w.records) == 0 {
            return nil
        }
        return w.flushToFile()
    }
    ```

    flushToFile is identical to BinaryEvtxWriter.flushToFile in cee-exporter (single-chunk write-on-close model). The stub flushChunkLocked behavior is preserved — Phase 10 fixes it.
