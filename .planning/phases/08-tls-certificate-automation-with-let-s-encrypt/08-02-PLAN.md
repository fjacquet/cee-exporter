// migrateListenConfig converts the legacy tls=true + cert_file/key_file pattern
// to tls_mode="manual" so that old config.toml files keep working after the
// Phase 8 upgrade.
func migrateListenConfig(cfg *ListenConfig) {
    if cfg.TLSMode != "" {
        return // explicit tls_mode set — no migration needed
    }
    if cfg.TLS && cfg.CertFile != "" {
        cfg.TLSMode = "manual"
        return
    }
    cfg.TLSMode = "off"
}
```

# ─────────────────────────────────────────────────────────────────────────────
# HTTP listener
# ─────────────────────────────────────────────────────────────────────────────
[listen]
addr = "0.0.0.0:12228"

# TLS mode — controls how the CEPA listener obtains its TLS certificate.
#
# IMPORTANT: Dell PowerStore CEPA sends events over plain HTTP only. TLS on
# port 12228 does NOT encrypt the PowerStore→exporter path. Use TLS here only
# if a TLS-capable reverse proxy sits between PowerStore and cee-exporter.
#
#   off         — No TLS (default). Plain HTTP on addr.
#   manual      — Load cert and key from cert_file / key_file.
#   acme        — Obtain and auto-renew certificate from Let's Encrypt via
#                 ACME TLS-ALPN-01. Requires port 443 reachable from the
#                 internet and acme_domains set. Needs CAP_NET_BIND_SERVICE
#                 on Linux (systemd unit adds it when this mode is used).
#   self-signed — Generate a runtime self-signed ECDSA cert. No files, no
#                 network. Useful for air-gapped or private-network deployments
#                 where TLS termination is desired but public CA is unavailable.
#
# Backward compat: the old tls=true + cert_file/key_file combination is
# automatically treated as tls_mode="manual".
#
tls_mode = "off"

# tls_mode = "manual" — provide paths to your certificate and private key:
# cert_file = "/etc/cee-exporter/tls/server.crt"
# key_file  = "/etc/cee-exporter/tls/server.key"

# tls_mode = "acme" — Let's Encrypt automatic certificate:
# acme_domains       = ["cepa.example.com"]
# acme_email         = "ops@example.com"
# acme_cache_dir     = "/var/cache/cee-exporter/acme"
# acme_challenge_addr = ":443"   # port must be 443 for TLS-ALPN-01

# tls_mode = "self-signed" — runtime-generated cert (no files needed):
# (no additional fields required)
```

Also update config.toml (the working config) — replace the [listen] section to add `tls_mode = "off"` as an explicit default, and add the new fields as comments to aid discoverability. Keep `tls = false` commented out for backward compat documentation.
